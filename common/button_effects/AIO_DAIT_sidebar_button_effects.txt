AIO_DAIT_Sidebar_Material_Button_Deactive = {
    potential = { always = yes }
    allow = {
        custom_tooltip = {
            fail_text = "AIO_DAIT_sidebar_change_faile_text"
            NOT = { check_variable = { which = AIO_DAIT_tab value = 1 } }
        }
    }
    effect = {
        hidden_effect = {
            set_variable = { which = AIO_DAIT_tab value = 1 }
            set_variable = { which = AIO_DAIT_page value = 1 }
        }
    }
}
AIO_DAIT_Sidebar_Material_Button_Active = {
    potential = { check_variable = { which = AIO_DAIT_tab value = 1 } }
    allow = { always = no }
}
