crisis_AIO_level_1 = {
	# REQUIREMENTS
	required_crisis_currency = 0

	# REWARDS
	perks = {
		menp_AIO_al_research
		menp_AIO_buildings_1
		menp_AIO_fleet_component_1_1
		menp_AIO_fleet_component_1_2
	}

	on_unlock = {
		hidden_effect = {
			begin_event_chain = {
				event_chain = AIO_crisis_events_chain
				target = root
			}
			add_event_chain_counter = {
				event_chain = "AIO_crisis_events_chain"
				counter = "AIO_crisis_events_chain"
				amount = 1
			}
		}
		country_event = { id = AIO_crisis_events.201 }
	}
}

crisis_AIO_level_5 = {
	# REQUIREMENTS
	allow = {
		custom_tooltip = {
			success_text = AIO_allinone_level_5_req_success_tooltip
			fail_text = AIO_allinone_level_5_req_fail_tooltip
			OR = {
				has_country_flag = AIO_finish_merge_cosmos_stage_6
				has_country_flag = AIO_empire_country
			}
		}
	}
	required_crisis_currency = 45000

	# REWARDS
	perks = {
		menp_AIO_mega_05
		menp_AIO_buildings_5
		menp_AIO_ascension_accelerate_5
	}

	on_unlock = {
		hidden_effect = {
			declare_crisis_effect = yes
			every_playable_country = {
				limit = {
					is_galactic_community_member = no
					NOT = {
						is_same_value = owner
						is_at_war_with = owner
					}
				}
				join_war_on_side = {
					war = event_target:crisis_war
					side = event_target:community_war_leader
				}
			}

			add_event_chain_counter = {
				event_chain = "AIO_crisis_events_chain"
				counter = "AIO_crisis_events_chain"
				amount = 1
			}
		}
		country_event = { id = AIO_crisis_events.205 }
	}
}
