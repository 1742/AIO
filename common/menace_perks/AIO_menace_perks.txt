# 舰船组件
menp_AIO_fleet_component_2 = {
	portrait = "GFX_menp_AIO_fleet_component_1_2_icon"

	on_unlock = {
        if = {
            limit = {
                is_machine_empire = yes
                country_uses_bio_ships = no
            }
            custom_tooltip = menp_AIO_ship_2_tt.machine.alloys
        }
        else_if = {
            limit = {
                is_machine_empire = yes
                country_uses_bio_ships = yes
            }
            custom_tooltip = menp_AIO_ship_2_tt.machine.food
        }
        else_if = {
            limit = {
                is_hive_empire = yes
                country_uses_bio_ships = no
            }
            custom_tooltip = menp_AIO_ship_2_tt.hive.alloys
        }
        else_if = {
            limit = {
                is_hive_empire = yes
                country_uses_bio_ships = yes
            }
            custom_tooltip = menp_AIO_ship_2_tt.machine.food
        }

		add_research_option = tech_nanite_transmutation
		add_research_option = tech_nanite_repair_system
		add_research_option = tech_nanite_autocannon
		add_research_option = tech_nanite_flak_batteries
	}
}
# 舰船
menp_AIO_ship_2 = {
	portrait = "GFX_menp_AIO_ship_2_icon"

			custom_tooltip = unlocks_ship_2_bio

		}
	}
}
