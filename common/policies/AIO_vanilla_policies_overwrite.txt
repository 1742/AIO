# 后勤政策
logistics_policy = {
	potential = { is_gestalt = yes }

	option = {
		name = "logistics_policy_gestalt"
        policy_flags = { logistics_policy_gestalt_default }

		potential = { is_gestalt = yes }

		on_enabled = { hidden_effect = { set_trade_conversions = { trade = 1 } } }

		on_disabled = { hidden_effect = { set_trade_conversions = { } } }

		AI_weight = { factor = 0 }
	}

    option = {
		name = "logistics_policy_gestalt_resource_with_food"
        policy_flags = { logistics_policy_gestalt_resource_with_food }

		potential = {
            has_modifier = AIO_DAIT_design_requirement_modifier
            is_gestalt = yes
            country_uses_food = yes
        }

		on_enabled = {
            set_trade_conversions = {
                trade = 1
                energy = 0.2
                minerals = 0.2
                food = 0.2
            }
        }

		on_disabled = { hidden_effect = { set_trade_conversions = { } } }

		AI_weight = { factor = 0 }
	}

    option = {
		name = "logistics_policy_gestalt_resource_without_food"
        policy_flags = { logistics_policy_gestalt_resource_without_food }

		potential = {
            has_modifier = AIO_DAIT_design_requirement_modifier
            is_gestalt = yes
            country_uses_food = no
        }

		on_enabled = {
            set_trade_conversions = {
                trade = 1
                energy = 0.3
                minerals = 0.3
            }
        }

		on_disabled = { hidden_effect = { set_trade_conversions = { } } }

		AI_weight = { factor = 0 }
	}

    option = {
		name = "logistics_policy_gestalt_industry_with_goods"
        policy_flags = { logistics_policy_gestalt_industry_with_goods }

		potential = {
            has_modifier = AIO_DAIT_design_requirement_modifier
            is_gestalt = yes
            country_uses_consumer_goods = yes
        }

		on_enabled = {
            set_trade_conversions = {
                trade = 1
                alloys = 0.2
                consumer_goods = 0.2
            }
        }

		on_disabled = { hidden_effect = { set_trade_conversions = { } } }

		AI_weight = { factor = 0 }
	}

    option = {
		name = "logistics_policy_gestalt_industry_without_goods"
        policy_flags = { logistics_policy_gestalt_industry_without_goods }

		potential = {
            has_modifier = AIO_DAIT_design_requirement_modifier
            is_gestalt = yes
            country_uses_consumer_goods = no
        }

		on_enabled = {
            set_trade_conversions = {
                trade = 1
                alloys = 0.3
            }
        }

		on_disabled = { hidden_effect = { set_trade_conversions = { } } }

		AI_weight = { factor = 0 }
	}

    option = {
		name = "logistics_policy_gestalt_data"
        policy_flags = { logistics_policy_gestalt_data }

		potential = {
            is_gestalt = yes
            has_modifier = AIO_DAIT_design_requirement_modifier
        }

		on_enabled = {
            set_trade_conversions = {
                trade = 1
                physics_research = 0.2
                society_research = 0.2
                engineering_research = 0.2
                unity = 0.15
            }
            add_modifier = { modifier = AIO_logistics_policy_gestalt_data days = -1 }
        }

		on_disabled = { hidden_effect = { set_trade_conversions = { } } }

		AI_weight = { factor = 0 }
	}
}
